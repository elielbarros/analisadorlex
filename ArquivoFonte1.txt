num val, i, resultado

resultado = 0

print "Digite um numero: "
read val

for i = 2 to (val / 2) step 1
{
   if val % i = 0 {
      resultado = resultado + 1
      break
   }
}

if resultado = 0
{
   print "O" + val + " eh um numero primo"
}
else
{
   print "O" + val + " NAO eh um numero primo"
}
